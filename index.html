    <script>

        // --- DATABASE LOGIC ---

        let db;

        const dbRequest = indexedDB.open("DriveLogDB", 1);


        dbRequest.onupgradeneeded = (e) => {

            db = e.target.result;

            db.createObjectStore("trips", { keyPath: "id", autoIncrement: true });

            db.createObjectStore("dailySummary", { keyPath: "date" });

            db.createObjectStore("settings", { keyPath: "id" });

        };


        dbRequest.onsuccess = (e) => {

            db = e.target.result;

            initializeApp();

        };


        // --- APP LOGIC ---

        let currentSettings = { id: 1, driverName: 'Himanshu', dailyRent: 1000, insurance: 50, fuelType: 'CNG', darkMode: false };

        const todayStr = new Date().toISOString().split('T')[0];


        async function initializeApp() {

            await loadSettings();

            renderDashboard();

            checkReminder();

            if (currentSettings.darkMode) document.body.classList.add('dark-mode');

        }


        function showScreen(screenId) {

            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

            document.getElementById(screenId).classList.add('active');

            document.querySelectorAll('nav button').forEach(b => b.classList.remove('active'));

            document.getElementById('nav-' + screenId.split('-')[1]).classList.add('active');

            

            if(screenId === 'screen-home') renderDashboard();

            if(screenId === 'screen-log') renderTrips();

            if(screenId === 'screen-pl') renderPL();

            if(screenId === 'screen-history') renderHistory();

            if(screenId === 'screen-analytics') renderAnalytics();

        }


        // --- SETTINGS ---

        async function loadSettings() {

            return new Promise(resolve => {

                const tx = db.transaction("settings", "readonly");

                const store = tx.objectStore("settings");

                const req = store.get(1);

                req.onsuccess = () => {

                    if (req.result) {

                        currentSettings = req.result;

                        document.getElementById('set-name').value = currentSettings.driverName;

                        document.getElementById('set-rent').value = currentSettings.dailyRent;

                        document.getElementById('set-ins').value = currentSettings.insurance;

                        document.getElementById('set-fuel').value = currentSettings.fuelType;

                        document.getElementById('set-dark').checked = currentSettings.darkMode;

                    }

                    resolve();

                };

            });

        }


        function saveSettings() {

            currentSettings = {

                id: 1,

                driverName: document.getElementById('set-name').value,

                dailyRent: parseFloat(document.getElementById('set-rent').value),

                insurance: parseFloat(document.getElementById('set-ins').value),

                fuelType: document.getElementById('set-fuel').value,

                darkMode: document.getElementById('set-dark').checked

            };

            const tx = db.transaction("settings", "readwrite");

            tx.objectStore("settings").put(currentSettings);

            showToast("Settings Updated");

        }


        function toggleDarkMode() {

            document.body.classList.toggle('dark-mode');

        }


        // --- TRIPS ---

        async function addTrip() {

            const trip = {

                date: todayStr,

                from: document.getElementById('trip-from').value,

                to: document.getElementById('trip-to').value,

                fare: parseFloat(document.getElementById('trip-fare').value) || 0,

                tip: parseFloat(document.getElementById('trip-tip').value) || 0,

                duration: parseInt(document.getElementById('trip-duration').value) || 0,

                waitingTime: parseInt(document.getElementById('trip-wait').value) || 0,

                paymentMode: document.getElementById('trip-pay').value,

                time: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })

            };


            const tx = db.transaction("trips", "readwrite");

            tx.objectStore("trips").add(trip);

            showToast("Trip Added");

            renderTrips();

            

            // Reset form

            document.getElementById('trip-from').value = '';

            document.getElementById('trip-to').value = '';

            document.getElementById('trip-fare').value = '';

            document.getElementById('trip-tip').value = '0';

        }


        async function renderTrips() {

            const tx = db.transaction("trips", "readonly");

            const store = tx.objectStore("trips");

            const req = store.getAll();

            req.onsuccess = () => {

                const todayTrips = req.result.filter(t => t.date === todayStr);

                const list = document.getElementById('trips-list');

                list.innerHTML = '<h4>Today\'s Trips</h4>';

                let total = 0;

                todayTrips.forEach((t, i) => {

                    total += (t.fare + t.tip);

                    list.innerHTML += `

                        <div class="card trip-item">

                            <div style="display:flex; justify-content:space-between">

                                <strong>${t.from} ➔ ${t.to}</strong>

                                <span class="currency">₹${t.fare + t.tip}</span>

                            </div>

                            <div class="trip-meta">

                                <span>${t.time} | ${t.duration}m | Wait: ${t.waitingTime}m</span>

                                <span class="badge">${t.paymentMode}</span>

                            </div>

                        </div>

                    `;

                });

                // Load daily header info

                const hReq = db.transaction("dailySummary", "readonly").objectStore("dailySummary").get(todayStr);

                hReq.onsuccess = () => {

                    if(hReq.result) {

                        document.getElementById('start-km').value = hReq.result.startKM || 0;

                        document.getElementById('end-km').value = hReq.result.endKM || 0;

                    }

                };

            };

        }


        // --- DASHBOARD & PL ---

        async function renderDashboard() {

            const trips = await getAllTrips();

            const todayTrips = trips.filter(t => t.date === todayStr);

            const gross = todayTrips.reduce((acc, t) => acc + t.fare + t.tip, 0);

            

            document.getElementById('home-date').innerText = new Date().toLocaleDateString('en-IN', { weekday: 'long', day: 'numeric', month: 'short' });

            document.getElementById('dash-gross').innerText = formatINR(gross);

            document.getElementById('dash-trips').innerText = todayTrips.length;

            

            // Calculate Net for dashboard

            const summary = await getSummary(todayStr);

            const net = gross - currentSettings.dailyRent - currentSettings.insurance - (summary.fuel || 0) - (summary.other || 0);

            document.getElementById('dash-net').innerText = formatINR(net);

            document.getElementById('dash-net').style.color = net >= 0 ? 'var(--accent)' : 'var(--loss)';

        }


        async function renderPL() {

            const trips = await getAllTrips();

            const todayTrips = trips.filter(t => t.date === todayStr);

            const gross = todayTrips.reduce((acc, t) => acc + t.fare + t.tip, 0);

            const summary = await getSummary(todayStr);


            document.getElementById('pl-gross').innerText = formatINR(gross);

            document.getElementById('pl-rent').innerText = `- ${formatINR(currentSettings.dailyRent)}`;

            document.getElementById('pl-ins').innerText = `- ${formatINR(currentSettings.insurance)}`;

            document.getElementById('pl-fuel-type').innerText = currentSettings.fuelType;

            document.getElementById('pl-fuel-input').value = summary.fuel || 0;

            document.getElementById('pl-other-input').value = summary.other || 0;


            const fuel = parseFloat(document.getElementById('pl-fuel-input').value) || 0;

            const other = parseFloat(document.getElementById('pl-other-input').value) || 0;

            const net = gross - currentSettings.dailyRent - currentSettings.insurance - fuel - other;


            document.getElementById('pl-net').innerText = formatINR(net);

            document.getElementById('pl-net').style.color = net >= 0 ? 'var(--accent)' : 'var(--loss)';


            // Metrics

            const km = (summary.endKM || 0) - (summary.startKM || 0);

            document.getElementById('pl-tot-km').innerText = km + " km";

            document.getElementById('pl-rate-km').innerText = km > 0 ? formatINR(gross/km) : '₹0';

            document.getElementById('pl-tot-wait').innerText = todayTrips.reduce((acc,t) => acc + t.waitingTime, 0) + "m";

            document.getElementById('pl-avg-fare').innerText = todayTrips.length > 0 ? formatINR(gross/todayTrips.length) : '₹0';

        }

        function saveDailyHeader(fuel = null, other = null) {

            const startKM = parseFloat(document.getElementById('start-km').value) || 0;

            const endKM = parseFloat(document.getElementById('end-km').value) || 0;

            

            const tx = db.transaction("dailySummary", "readwrite");

            const store = tx.objectStore("dailySummary");

            const req = store.get(todayStr);

            req.onsuccess = () => {

                const data = req.result || { date: todayStr };

                data.startKM = startKM;

                data.endKM = endKM;

                if(fuel !== null) data.fuel = fuel;

                if(other !== null) data.other = other;

                store.put(data);

            };

        }

        async function renderHistory() {

            const trips = await getAllTrips();

            const summaries = await getAllSummaries();

            const dates = [...new Set(trips.map(t => t.date))].sort().reverse();

            

            const list = document.getElementById('history-list');

            list.innerHTML = '';

            

            dates.forEach(d => {

                const dayTrips = trips.filter(t => t.date === d);

                const daySummary = summaries.find(s => s.date === d) || {};

                const gross = dayTrips.reduce((acc, t) => acc + t.fare + t.tip, 0);

                const net = gross - currentSettings.dailyRent - currentSettings.insurance - (daySummary.fuel || 0) - (daySummary.other || 0);

                

                const colorClass = net > 0 ? 'var(--accent)' : 'var(--loss)';

                

                list.innerHTML += `

                    <div class="card" onclick="alert('Viewing full details for ${d}')">

                        <div style="display:flex; justify-content:space-between; align-items:center">

                            <strong>${new Date(d).toLocaleDateString('en-IN', {day:'2-digit', month:'short'})}</strong>

                            <span style="color:${colorClass}; font-weight:bold">${formatINR(net)} Net</span>

                        </div>

                        <div class="trip-meta">

                            <span>${dayTrips.length} Trips | Gross: ${formatINR(gross)}</span>

                        </div>

                    </div>

                `;

            });

        }

        async function renderAnalytics() {

            const trips = await getAllTrips();

            const summaries = await getAllSummaries();

            const last7Days = [];

            for(let i=6; i>=0; i--) {

                const d = new Date();

                d.setDate(d.getDate() - i);

                last7Days.push(d.toISOString().split('T')[0]);

            }


            const dailyNet = last7Days.map(d => {

                const dayTrips = trips.filter(t => t.date === d);

                const daySummary = summaries.find(s => s.date === d) || {};

                const gross = dayTrips.reduce((acc, t) => acc + t.fare + t.tip, 0);

                return gross - currentSettings.dailyRent - currentSettings.insurance - (daySummary.fuel || 0) - (daySummary.other || 0);

            });


            // Payment Split

            const modes = { UPI: 0, Cash: 0, Card: 0 };

            trips.forEach(t => modes[t.paymentMode]++);


            // Charts

            new Chart(document.getElementById('earningsChart'), {

                type: 'line',

                data: {

                    labels: last7Days.map(d => d.split('-')[2]),

                    datasets: [{ label: 'Daily Net Profit (₹)', data: dailyNet, borderColor: '#1A3C5E', tension: 0.1 }]

                }

            });


            new Chart(document.getElementById('paymentChart'), {

                type: 'doughnut',

                data: {

                    labels: ['UPI', 'Cash', 'Card'],

                    datasets: [{ data: [modes.UPI, modes.Cash, modes.Card], backgroundColor: ['#2ECC71', '#F1C40F', '#3498DB'] }]

                },

                options: { plugins: { title: { display: true, text: 'Payment Mode Split' } } }

            });

        }
